    **app/(public)/strefa-dzialacza/lista-obecnosci/page.tsx** — 'use client' (needs localStorage + CSV export). Imports spotkania from data/spotkania.ts. Renders meeting list with checkboxes. Attendance state persisted in localStorage under key 'ssuew-attendance'. CSV export button:
    ```typescript
    'use client'
    import { useState, useEffect } from 'react'
    import Link from 'next/link'
    import { spotkania } from '@/data/spotkania'
    import type { Spotkanie } from '@/data/spotkania'

    const STORAGE_KEY = 'ssuew-attendance'

    function exportCSV(meetings: Spotkanie[], attendance: Record<string, boolean>) {
      const cell = (v: string) => `"${v.replace(/"/g, '""')}"`
      const rows = meetings.map(m => [
        cell(m.date),
        cell(m.title),
        cell(attendance[m.id] ? 'obecny' : 'nieobecny'),
      ].join(','))
      const csv = ['Data,Spotkanie,Obecność', ...rows].join('\n')
      const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' })
      const url = URL.createObjectURL(blob)
      const a = document.createElement('a')
      a.href = url
      a.download = 'lista-obecnosci.csv'
      a.click()
      URL.revokeObjectURL(url)
    }

    export default function ListaObecnosci() {
      const [attendance, setAttendance] = useState<Record<string, boolean>>({})

      useEffect(() => {
        try {
          const stored = localStorage.getItem(STORAGE_KEY)
          if (stored) setAttendance(JSON.parse(stored))
        } catch {}
      }, [])

      function toggle(id: string) {
        const next = { ...attendance, [id]: !attendance[id] }
        setAttendance(next)
        try { localStorage.setItem(STORAGE_KEY, JSON.stringify(next)) } catch {}
      }

      const presentCount = Object.values(attendance).filter(Boolean).length

      return (
        <main>
          <section className="pt-36 pb-8 bg-gradient-to-br from-ssuew-gray-100 to-white">
            <div className="max-w-[1200px] mx-auto px-6">
              <h1 className="font-display text-display-xl text-ssuew-black mb-2">Lista obecności</h1>
              <p className="text-base text-ssuew-gray-600">
                Zaznaczono: {presentCount} / {spotkania.length} spotkań
              </p>
            </div>
          </section>

          <section className="py-16 bg-white">
            <div className="max-w-[1200px] mx-auto px-6">
              <div className="max-w-2xl flex flex-col gap-3">
                {spotkania.map((s) => (
                  <label
                    key={s.id}
                    className="flex items-center gap-4 p-4 rounded-xl border border-ssuew-gray-200 cursor-pointer hover:border-primary transition-colors duration-200"
                  >
                    <input
                      type="checkbox"
                      checked={!!attendance[s.id]}
                      onChange={() => toggle(s.id)}
                      className="w-5 h-5 accent-primary shrink-0"
                    />
                    <div>
                      <p className="font-bold text-ssuew-black text-[0.9rem]">{s.title}</p>
                      <p className="text-[0.8rem] text-ssuew-gray-600">{s.date}</p>
                    </div>
                  </label>
                ))}
              </div>

              <div className="mt-8 flex gap-4 flex-wrap">
                <button
                  onClick={() => exportCSV(spotkania, attendance)}
                  className="inline-flex items-center gap-2 font-bold rounded-full px-8 py-3.5 text-base bg-primary text-white hover:bg-primary-dark transition-colors duration-200"
                >
                  Pobierz CSV
                </button>
                <Link
                  href="/strefa-dzialacza/dashboard"
                  className="inline-flex items-center gap-2 px-5 py-2 rounded-full border-2 border-ssuew-gray-200 text-ssuew-gray-600 text-[0.85rem] font-bold hover:border-primary hover:text-primary transition-all duration-200"
                >
                  ← Dashboard
                </Link>
              </div>
            </div>
          </section>
        </main>
      )
    }
    ```

    ```typescript
    'use client'
    import { useState, useRef, useEffect } from 'react'
    import { useChat } from 'ai/react'   // v4 import path — NOT @ai-sdk/react

    export function ChatWidget() {
      const [open, setOpen] = useState(false)
      const bottomRef = useRef<HTMLDivElement>(null)
      const { messages, input, handleInputChange, handleSubmit, isLoading, error } = useChat({
        api: '/api/chat',
      })

      useEffect(() => {
        bottomRef.current?.scrollIntoView({ behavior: 'smooth' })
      }, [messages])

      return (
        <div className="fixed bottom-6 right-6 z-50 flex flex-col items-end gap-3">
          {open && (
            <div className="w-80 sm:w-96 bg-white rounded-2xl shadow-2xl border border-ssuew-gray-200 flex flex-col overflow-hidden">
              {/* Header */}
              <div className="bg-primary px-4 py-3 flex items-center justify-between">
                <div>
                  <p className="font-bold text-white text-[0.9rem]">Asystent SSUEW</p>
                  <p className="text-[0.7rem] text-white/80">Zadaj pytanie o samorząd</p>
                </div>
                <button
                  onClick={() => setOpen(false)}
                  aria-label="Zamknij chat"
                  className="text-white/80 hover:text-white transition-colors text-xl leading-none"
                >
                  ×
                </button>
              </div>

              {/* Messages */}
              <div className="flex-1 overflow-y-auto p-4 flex flex-col gap-3 max-h-72">
                {messages.length === 0 && (
                  <p className="text-[0.8rem] text-ssuew-gray-600 text-center pt-4">
                    Cześć! Jestem asystentem SSUEW. Jak mogę pomóc?
                  </p>
                )}
                {messages.map((m) => (
                  <div
                    key={m.id}
                    className={`text-[0.85rem] leading-relaxed rounded-xl px-3 py-2 max-w-[85%] ${
                      m.role === 'user'
                        ? 'bg-primary text-white self-end'
                        : 'bg-ssuew-gray-100 text-ssuew-black self-start'
                    }`}
                  >
                    {m.content}
                  </div>
                ))}
                {isLoading && (
                  <div className="bg-ssuew-gray-100 text-ssuew-gray-600 text-[0.85rem] rounded-xl px-3 py-2 self-start">
                    Piszę...
                  </div>
                )}
                {error && (
                  <div className="bg-red-50 text-red-600 text-[0.8rem] rounded-xl px-3 py-2 self-start">
                    Chatbot jest niedostępny. Skontaktuj się przez{' '}
                    <a href="/formularz" className="underline">formularz</a>.
                  </div>
                )}
                <div ref={bottomRef} />
              </div>

              {/* Input */}
              <form onSubmit={handleSubmit} className="border-t border-ssuew-gray-200 flex">
                <input
                  value={input}
                  onChange={handleInputChange}
                  placeholder="Napisz wiadomość..."
                  disabled={isLoading}
                  className="flex-1 px-4 py-3 text-[0.85rem] text-ssuew-black bg-white focus:outline-none placeholder-ssuew-gray-400"
                />
                <button
                  type="submit"
                  disabled={isLoading || !input.trim()}
                  aria-label="Wyślij"
                  className="px-4 py-3 text-primary font-bold hover:text-primary-dark disabled:text-ssuew-gray-400 transition-colors text-[0.85rem]"
                >
                  →
                </button>
              </form>
            </div>
          )}

          {/* Toggle button */}
          <button
            onClick={() => setOpen(!open)}
            aria-label={open ? 'Zamknij asystenta' : 'Otwórz asystenta SSUEW'}
            className="w-14 h-14 rounded-full bg-primary text-white shadow-lg hover:bg-primary-dark transition-all duration-200 flex items-center justify-center text-2xl"
          >
            {open ? '×' : '💬'}
          </button>
        </div>
      )
    }
    ```
